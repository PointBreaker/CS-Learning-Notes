import random
n = 1000000

dis = [0] * 20

for i in range(n):
    blue = 0
    red = 0
    for j in range(20):
        if random.random() >= 0.3:
            blue += 1
        else:
            red += 1
    dis[red] += 1

print([i/n for i in dis])

from math import comb
ans = []
for i in range(20):
    ans.append(comb(20, i) * (0.3 ** i) * (0.7 ** (20 - i)))
print(ans)
